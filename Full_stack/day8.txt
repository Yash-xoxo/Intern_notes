function fetchData() {
  // Create an AJAX object
  let xhr = new XMLHttpRequest();
  xhr.open("GET", "db.txt", true);
  xhr.onload = function() {
    if (xhr.status === 200) {
      document.getElementById("outputDiv").innerHTML = xhr.responseText;
    }
  };
  xhr.send();
}
